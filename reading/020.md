## Data flow overview

```
handleRead()
    → recv() raw bytes into read_buffer_
    → RequestParser::feed() — incremental, handles chunked + multipart
        → when complete: RequestParser::isComplete() == true
        → extract HttpRequest
    → ResponseBuilder::build(request) — reads static file, builds response
        → store into write_buffer_
    → reactor_.enableWrite(this)

handleWrite()
    → send() from write_buffer_
    → when drained: reactor_.disableWrite(this)
    → if Connection: close → reactor_.deleteHandler(fd_)
```

```cpp
#ifndef HTTPREQUEST_HPP
#define HTTPREQUEST_HPP

#include <string>
#include <map>
#include <vector>

class HttpRequest
{
public:
    enum Method { GET, POST, DELETE, UNKNOWN };

    HttpRequest()
        : method_(UNKNOWN)
        , content_length_(0)
        , is_chunked_(false)
        , is_multipart_(false)
    {}

    // setters — used by RequestParser
    void setMethod(Method m)                              { method_ = m; }
    void setUri(const std::string &uri)                   { uri_ = uri; }
    void setVersion(const std::string &v)                 { version_ = v; }
    void setHeader(const std::string &k, const std::string &v) { headers_[k] = v; }
    void setBody(const std::vector<char> &body)           { body_ = body; }
    void setContentLength(size_t n)                       { content_length_ = n; }
    void setChunked(bool v)                               { is_chunked_ = v; }
    void setMultipart(bool v)                             { is_multipart_ = v; }
    void setMultipartBoundary(const std::string &b)       { multipart_boundary_ = b; }

    // getters — used by ResponseBuilder
    Method                                   getMethod()            const { return method_; }
    const std::string&                       getUri()               const { return uri_; }
    const std::string&                       getVersion()           const { return version_; }
    const std::map<std::string,std::string>& getHeaders()           const { return headers_; }
    const std::vector<char>&                 getBody()              const { return body_; }
    size_t                                   getContentLength()     const { return content_length_; }
    bool                                     isChunked()            const { return is_chunked_; }
    bool                                     isMultipart()          const { return is_multipart_; }
    const std::string&                       getMultipartBoundary() const { return multipart_boundary_; }

    std::string getHeader(const std::string &key) const
    {
        std::map<std::string,std::string>::const_iterator it = headers_.find(key);
        return it != headers_.end() ? it->second : "";
    }

private:
    Method                             method_;
    std::string                        uri_;
    std::string                        version_;
    std::map<std::string, std::string> headers_;
    std::vector<char>                  body_;
    size_t                             content_length_;
    bool                               is_chunked_;
    bool                               is_multipart_;
    std::string                        multipart_boundary_;
};

#endif
```

## RequestParser.hpp
```cpp
#ifndef REQUESTPARSER_HPP
#define REQUESTPARSER_HPP

#include "HttpRequest.hpp"
#include <vector>
#include <string>

class RequestParser
{
public:
    enum State {
        PARSE_REQUEST_LINE,
        PARSE_HEADERS,
        PARSE_BODY_IDENTITY,   // plain Content-Length body
        PARSE_BODY_CHUNKED,    // Transfer-Encoding: chunked
        PARSE_COMPLETE,
        PARSE_ERROR
    };

    RequestParser()
        : state_(PARSE_REQUEST_LINE)
        , bytes_remaining_(0)
        , current_chunk_size_(0)
        , chunk_state_(CHUNK_SIZE)
    {}

    // Feed raw bytes incrementally — call every time recv() returns data
    // Returns false on parse error
    bool feed(const char *data, size_t len);

    bool isComplete() const { return state_ == PARSE_COMPLETE; }
    bool hasError()   const { return state_ == PARSE_ERROR; }

    // Call only when isComplete() — extract the finished request
    const HttpRequest& getRequest() const { return request_; }

    void reset()
    {
        state_             = PARSE_REQUEST_LINE;
        bytes_remaining_   = 0;
        current_chunk_size_= 0;
        chunk_state_       = CHUNK_SIZE;
        raw_buffer_.clear();
        request_           = HttpRequest();
    }

private:
    bool parseRequestLine(const std::string &line);
    bool parseHeaderLine(const std::string &line);
    bool finalizeHeaders();               // called after blank line
    bool parseBodyIdentity();
    bool parseBodyChunked();

    std::string extractLine();            // pulls one \r\n line from raw_buffer_
    bool        hasLine() const;

    State            state_;
    HttpRequest      request_;
    std::vector<char> raw_buffer_;        // accumulates recv() bytes

    size_t           bytes_remaining_;    // for identity body

    // chunked decoding state
    enum ChunkState { CHUNK_SIZE, CHUNK_DATA, CHUNK_TRAILER };
    ChunkState       chunk_state_;
    size_t           current_chunk_size_;
};

#endif
```

## ConnectionHandler.cpp — full integration
```cpp
#include "ConnectionHandler.hpp"
#include "Reactor.hpp"
#include <sys/socket.h>
#include <cerrno>
#include <cstring>
#include <iostream>

ConnectionHandler::ConnectionHandler(int fd, Reactor &reactor,
                                     const std::string &root_dir)
    : IEventHandler(IEventHandler::CONNECTION)
    , fd_(fd)
    , reactor_(reactor)
    , builder_(root_dir)
    , write_offset_(0)
{}

ConnectionHandler::~ConnectionHandler()
{
    if (fd_ != -1)
        close(fd_);
}

int ConnectionHandler::getFd() const { return fd_; }

// ── READ ──────────────────────────────────────────────────────────────────────
void ConnectionHandler::handleRead()
{
    char buf[4096];

    while (true)  // drain the socket — it's non-blocking
    {
        ssize_t n = recv(fd_, buf, sizeof(buf), 0);

        if (n == 0)  // client closed connection
        {
            reactor_.deleteHandler(fd_);
            return;
        }
        if (n < 0)
        {
            if (errno == EAGAIN || errno == EWOULDBLOCK)
                break;  // no more data right now — wait for next EPOLLIN
            reactor_.deleteHandler(fd_);
            return;
        }

        if (!parser_.feed(buf, static_cast<size_t>(n)))
        {
            // parse error — send 400 and close
            HttpResponse err = builder_.build(HttpRequest()); // will produce 400
            write_buffer_ = err.serialize();
            write_offset_ = 0;
            reactor_.enableWrite(this);
            return;
        }

        if (parser_.isComplete())
        {
            processRequest();
            return;
        }
    }
}

// ── PROCESS ───────────────────────────────────────────────────────────────────
void ConnectionHandler::processRequest()
{
    HttpRequest  request  = parser_.getRequest();
    HttpResponse response = builder_.build(request);

    write_buffer_ = response.serialize();
    write_offset_ = 0;

    reactor_.enableWrite(this);  // arm EPOLLOUT — handleWrite() will send it

    parser_.reset();             // ready for next request (keep-alive)
}

// ── WRITE ─────────────────────────────────────────────────────────────────────
void ConnectionHandler::handleWrite()
{
    while (write_offset_ < write_buffer_.size())
    {
        ssize_t n = send(fd_,
                         write_buffer_.data() + write_offset_,
                         write_buffer_.size() - write_offset_,
                         0);
        if (n < 0)
        {
            if (errno == EAGAIN || errno == EWOULDBLOCK)
                return;  // kernel buffer full — wait for next EPOLLOUT
            reactor_.deleteHandler(fd_);
            return;
        }
        write_offset_ += static_cast<size_t>(n);
    }

    // Buffer fully sent
    write_buffer_.clear();
    write_offset_ = 0;
    reactor_.disableWrite(this);  // remove EPOLLOUT — avoid busy-loop

    if (shouldClose())
        reactor_.deleteHandler(fd_);
}

// ── ERROR ─────────────────────────────────────────────────────────────────────
void ConnectionHandler::handleError()
{
    reactor_.deleteHandler(fd_);
}

// ── HELPERS ───────────────────────────────────────────────────────────────────
bool ConnectionHandler::shouldClose() const
{
    // Close if HTTP/1.0 or explicit "Connection: close" header
    std::string conn = parser_.getRequest().getHeader("Connection");
    std::string ver  = parser_.getRequest().getVersion();
    return (ver == "HTTP/1.0" || conn == "close");
}
```

## Key design decisions to remember

| Decision | Why |
|---|---|
| `write_offset_` instead of erasing buffer | Erasing from front of `vector` is O(n) every send — offset is O(1) |
| Drain loop in `handleRead` | Non-blocking fd may have multiple chunks ready — one `EPOLLIN` can carry many |
| `parser_.reset()` after `processRequest()` | Enables HTTP keep-alive — same connection handles multiple requests |
| `enableWrite` only when response is ready | Never register `EPOLLOUT` permanently — epoll will busy-loop |
| `shouldClose()` checked after buffer drained | Don't close mid-send; finish the response first |

When you're ready to add CGI, the `CGIHandler` will follow the exact same pattern — `handleRead` reads from the CGI pipe, `handleWrite` forwards to the client — and the `cgi_done_` flag will tell `ConnectionHandler` when the pipe is exhausted.
